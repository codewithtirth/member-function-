#include<iostream>
using namespace std;

class member
{
     public:
        void inside()
        {
        cout<<"This is inside member function"<<endl;
        
        }
        void outside();
};
     void member::outside()
    {
        cout<<"This is outside member function"<<endl;
        
    }
int main()
{
          member m;
        m.inside();
        m.outside();
        
        return 0;
}
